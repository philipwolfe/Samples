Invoked if we ever returned true from the // LoadPostData method (indicates that we want a change notification raised). Since we // always return false, this method is just a no-op. } protected override void Render(HtmlTextWriter output) { output.Write("<h3>Value: <input name=" + this.UniqueID + " type=text value=" + this.Value + "> </h3>"); } } }
