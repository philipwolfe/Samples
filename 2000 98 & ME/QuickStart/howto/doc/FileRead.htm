StringBuilder output = new StringBuilder(); FileStream fs = File.OpenRead (Server.MapPath("datafile.xml")); StreamReader sr = new StreamReader(fs); String line = sr.ReadLine(); while (line != null) { output.Append(line + "\n"); line = sr.ReadLine(); } fs.Close();
